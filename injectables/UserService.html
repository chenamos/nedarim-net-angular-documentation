import { Injectable } from '@angular/core';
import { HttpClient, HttpErrorResponse, HttpHeaders } from '@angular/common/http';
import { Observable, throwError } from 'rxjs';
import { catchError } from 'rxjs/operators';
import { AppUser } from './models/app-user.model';
import { Role } from './models/role.model';
import { Endpoints } from './constants/base-url.constants';  // Import the correct Endpoints constant

@Injectable({
  providedIn: 'root'
})
export class UserService {

  constructor(private http: HttpClient) { }

// Fetch all users
getAllUsers(): Observable<AppUser[]> {
  return this.http.get<AppUser[]>(Endpoints.user.getAllUsers).pipe(
    catchError(this.handleError)
  );
}
  // Register a new user
  registerUser(user: AppUser): Observable<any> {
    return this.http.post(Endpoints.members.add, user).pipe(
      catchError(this.handleError)
    );
  }

  // Assign roles to a user
  assignRoles(userId: number, roleNames: string[]): Observable<any> {
    return this.http.post(Endpoints.user.assignRoles, { userId, roleNames }).pipe(
      catchError(this.handleError)
    );
  }
  

  // Fetch all roles
  getRoles(): Observable<Role[]> {
    return this.http.get<Role[]>(Endpoints.user.roles).pipe(
      catchError(this.handleError)
    );
  }
  
  // Fetch current user's permissions/roles
  getCurrentUserPermissions(): Observable<string[]> {
    return this.http.get<string[]>(Endpoints.user.permissions).pipe(
      catchError(this.handleError)
    );
  }

  // Fetch the current user's clientId and other information
  getCurrentUserInfo(): Observable<any> {
    return this.http.get<any>(Endpoints.user.info).pipe(
      catchError(this.handleError)
    );
  }

  // Upload Excel file to process
  uploadExcelFile(file: File): Observable<any> {
    const formData = new FormData();
    formData.append('file', file);

    // Replace this with the logic to retrieve your JWT token
    const token = 'your-jwt-token-here'; 

    const headers = new HttpHeaders({
      'Authorization': `Bearer ${token}`
    });

    return this.http.post(Endpoints.excel.upload, formData, { headers }).pipe(
      catchError(this.handleError)
    );
  }

  // Upload Expenditures Excel file
  uploadExpendituresExcelFile(file: File): Observable<any> {
    const formData = new FormData();
    formData.append('file', file);
  
    const token = 'your-jwt-token-here'; // Retrieve or generate the JWT token as needed
  
    const headers = new HttpHeaders({
      'Authorization': `Bearer ${token}`
    });
  
    return this.http.post(Endpoints.excel.uploadExpenditures, formData, { headers }).pipe(
      catchError(this.handleError)
    );
  }

  // Error handling
  private handleError(error: HttpErrorResponse) {
    let errorMessage = 'An unknown error occurred!';
    if (error.error instanceof ErrorEvent) {
      // Client-side or network error
      errorMessage = `Error: ${error.error.message}`;
    } else {
      // Backend error
      errorMessage = `Server returned code: ${error.status}, error message is: ${error.message}`;
    }
    return throwError(errorMessage);
  }
}
