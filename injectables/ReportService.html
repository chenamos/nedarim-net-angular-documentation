import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { BehaviorSubject, Observable, of, catchError, map, tap, shareReplay } from 'rxjs';
import { Endpoints } from './constants/base-url.constants';  

@Injectable({
  providedIn: 'root'
})
export class ReportService {
  private cache: any = {};  // Cache to store data temporarily
  private filteredDataSubject = new BehaviorSubject<any[]>([]); 

  constructor(private http: HttpClient) {}

  // Fetch data if it's not already in the cache
  fetchDataIfNeeded(): Observable<any[]> {
    if (!this.cache['data']) {
      return this.http.post<any[]>(Endpoints.donations.getAll, {}).pipe(
        tap(data => {
          this.cache['data'] = data;  // Cache the data
          console.log('Data fetched and cached', data);  // Log fetched data
        }),
        shareReplay(1)  // Share the replay of this observable to ensure only one request is made
      );
    }
    console.log('Using cached data', this.cache['data']);  // Log usage of cached data
    return of(this.cache['data']);  // Return cached data if available
  }

  // Method for getting filtered data based on provided filters
  getFilteredReports(filterData: any): Observable<any[]> {
    console.log('Filtering data with params', filterData);  // Confirm the filter received
  
    return this.fetchDataIfNeeded().pipe(
      tap(data => console.log('Pre-filter data count', data.length)),  // Log pre-filter data count
      map(data => {
        const filtered = data.filter(item => {
          let matches = true;

          // Apply filters and debug logs for each field
          if (filterData.donorName) {
            console.log('Filtering by donorName:', filterData.donorName);
            matches = matches && item.donorName?.toLowerCase() === filterData.donorName.toLowerCase();
          }
          if (filterData.donatedFor) {
            console.log('Filtering by donatedFor:', filterData.donatedFor);
            matches = matches && item.donatedFor?.toLowerCase() === filterData.donatedFor.toLowerCase();
          }
          if (filterData.parashaName) {
            console.log('Filtering by parashaName:', filterData.parashaName);
            matches = matches && item.parashaName?.toLowerCase() === filterData.parashaName.toLowerCase();
          }

          return matches;
        });
        console.log('Filtered data result count', filtered.length);  // Log filtered data count
        return filtered;
      }),
      catchError(error => {
        console.error('Error filtering data:', error);
        return of([]);  // Handle errors gracefully
      })
    );
  }

  // Set filtered data into the BehaviorSubject
  setFilteredData(data: any[]): void {
    this.filteredDataSubject.next(data);
    console.log('Filtered data set in BehaviorSubject', this.filteredDataSubject.value);
  }

  // Return filtered data as Observable
  getFilteredData(): Observable<any[]> {
    return this.filteredDataSubject.asObservable();
  }

  // Helper method to get unique donor names for dropdown
  getDonorNames(): Observable<string[]> {
    return this.fetchDataIfNeeded().pipe(
      map(data => data.map(item => item.donorName)
        .filter((value, index, self) => value && self.indexOf(value) === index))  // Ensure unique values
    );
  }

  // Helper method to get unique donatedFor values for dropdown
  getDonatedFor(): Observable<string[]> {
    return this.fetchDataIfNeeded().pipe(
      map(data => {
        const donatedForValues = data.map(item => item.donatedFor)
          .filter((value, index, self) => value && self.indexOf(value) === index);

        console.log('DonatedFor values fetched:', donatedForValues);  // Log the fetched values
        return donatedForValues;
      })
    );
  }

  // Helper method to get unique parasha names for dropdown
  getParshaNames(): Observable<string[]> {
    return this.fetchDataIfNeeded().pipe(
      map(data => data.map(item => item.parashaName)
        .filter((value, index, self) => value && self.indexOf(value) === index))  // Ensure unique values
    );
  }
}
