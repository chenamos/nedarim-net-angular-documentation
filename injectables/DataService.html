import { Injectable } from '@angular/core';
import { HttpClient, HttpErrorResponse, HttpHeaders, HttpParams } from '@angular/common/http';
import { Observable, catchError, map, of, switchMap, tap, throwError } from 'rxjs';
import { Endpoints } from './constants/base-url.constants';  // Import the Endpoints

@Injectable({
  providedIn: 'root'
})
export class DataService {

  constructor(private http: HttpClient) {}

  getAllData(): Observable<any[]> {
    return this.http.post<any[]>(Endpoints.donations.getAll, {}).pipe(  // Use post instead of get
      catchError(this.handleError<any[]>('getAllData', []))
    );
  }
getExpendituresData(): Observable<any[]> {
  return this.http.post<any[]>(Endpoints.financial.getExpenditures, {}).pipe(
    tap(data => console.log('Received expenditures:', data)),
    catchError(this.handleError<any[]>('getExpendituresData', []))
  );
}

   // Fetch paginated donation data
   getPaginatedData(page: number, itemsPerPage: number): Observable<{ items: any[], total: number }> {
    const params = new HttpParams()
      .set('page', page.toString())
      .set('size', itemsPerPage.toString());

    return this.http.post<{ items: any[], total: number }>(Endpoints.donations.getPaginated, {}, { params }).pipe(
      catchError(this.handleError<{ items: any[], total: number }>('getPaginatedData', { items: [], total: 0 }))
    );
  }

// Fetch financial summary
getFinancialSummary(): Observable<{ total_paid: number; total_expenses: number; net_balance: number }> {
  return this.http.post<{ total_paid: number; total_expenses: number; net_balance: number }>(
    Endpoints.donations.financialSummary,
    {}
  ).pipe(
    map((response: any) => ({
      total_paid: response.total_paid || 0,
      total_expenses: response.total_expenses || 0,
      net_balance: response.net_balance || 0
    })),
    catchError(this.handleError('getFinancialSummary', { total_paid: 0, total_expenses: 0, net_balance: 0 }))
  );
}


addExpenditure(data: any): Observable<any> {
  return this.http.post(Endpoints.financial.addExpenditure, data, {
    headers: new HttpHeaders({
      'Content-Type': 'application/json'
    }),
    responseType: 'text'
  }).pipe(
    tap(response => console.log('Expenditure added:', response)),
    catchError(err => {
      console.error('Error adding expenditure:', err);
      return throwError(() => new Error('Failed to add expenditure'));
    })
  );
}


  addNewRecord(data: any): Observable<any> {
    return this.http.post(Endpoints.donations.add, data, {
      headers: new HttpHeaders({ 'Content-Type': 'application/json' })
    }).pipe(
      catchError(this.handleError('addNewRecord'))
    );
  }

  

  addPaymentRecord(paymentData: any): Observable<any> {
    return this.http.post(Endpoints.credit.processPayment, paymentData, {
      headers: new HttpHeaders({
        'Content-Type': 'application/json'  // Sending JSON data to the backend
      }),
      responseType: 'text'  // Expecting a plain text response from the backend
    });
  }
  
  
   // Process payment (without changing the method)
   processPayment(paymentData: any): Observable<any> {
    console.log('Preparing to process payment...');

    const formattedExpDate = this.formatExpirationDate(paymentData.expDate);

    const paymentRequest = {
      cardNumber: paymentData.cardNumber,
      cvv: paymentData.cvv,
      expDate: formattedExpDate,
      transactionSum: paymentData.transactionSum,
      pledges: paymentData.pledges
    };

    console.log('Payment request data:', paymentRequest);

    return this.http.post(`${Endpoints.credit.processPayment}`, paymentRequest, {
      headers: new HttpHeaders({
        'Content-Type': 'application/json'
      }),
      observe: 'response',
      responseType: 'text'  // Explicitly set to 'text' in case the backend returns text
    }).pipe(
      tap(response => {
        console.log('Raw response:', response);
        if (response.status === 200) {
          alert('Payment processed successfully!');
        } else {
          console.warn('Unexpected response status:', response.status);
        }
      }),
      catchError((err: HttpErrorResponse) => {
        console.error('Error during payment processing:', err);
        if (err.status === 400) {
          const backendMessage = err.error ? err.error : 'An error occurred during payment processing.';
          alert(`Payment error: ${backendMessage}`);
        } else {
          alert('Unexpected error occurred.');
        }
        return throwError(() => new Error('Payment processing failed'));
      })
    );
  }
  /**
   * Helper method to format expiration date from MM/YYYY to MMYY.
   */
  formatExpirationDate(date: string): string {
    console.log('Expiration date before formatting:', date); // Add this for debugging
    if (!date || date.indexOf('/') === -1) {
      return date;
    }

    const [month, year] = date.split('/');
    return `${month}${year.slice(-2)}`;
  }

  /**
   * Handles HTTP operation failures.
   * @param operation - name of the operation that failed
   * @param result - optional value to return as the observable result
   */
  private handleError<T>(operation = 'operation', result?: T) {
    return (error: any): Observable<T> => {
      console.error(`${operation} failed: ${error.message}`);  // log to console
      return of(result as T);  // Let the app keep running by returning an empty result.
    };
  }
}
