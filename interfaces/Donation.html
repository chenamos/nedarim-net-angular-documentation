import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { MatDialog } from '@angular/material/dialog';
import { Observable, of, catchError, tap, shareReplay, BehaviorSubject, map } from 'rxjs';
import { AddMemberFormComponent } from './add-member-form/add-member-form.component';
import { Endpoints } from './constants/base-url.constants';  // Import Endpoints constant

@Injectable({
  providedIn: 'root'
})
export class MemberService {
  private memberCache = new BehaviorSubject<MemberResponse[]>([]);
  private donateCache = new BehaviorSubject<any[]>([]);

  constructor(private http: HttpClient, private dialog: MatDialog) {}

  fetchMembersIfNeeded(): Observable<MemberResponse[]> {
    this.memberCache.next([]); // Clear cache before fetching
    return this.http.post<MemberResponse[]>(Endpoints.members.details, {}).pipe(
      tap(data => {
        console.log('Fetched members:', data); // Log members
        this.memberCache.next(data);
      }),
      catchError(error => {
        console.error('Error fetching member data:', error);
        return of([]); // Return an empty array if there's an error
      }),
      shareReplay(1)
    );
  }

  getAllData(): Observable<MemberResponse[]> {
    const requestData = {}; // Modify if needed
    return this.http.post<any[]>(Endpoints.donations.getAll, requestData, {
      headers: new HttpHeaders({
        'Content-Type': 'application/json'
      })
    }).pipe(
      catchError(this.handleError<any[]>('getAllData', []))  // Handle errors gracefully
    );
  }

  private handleError<T>(operation = 'operation', result?: T) {
    return (error: any): Observable<T> => {
      console.error(`${operation} failed: ${error.message}`);  // log to console
      return of(result as T);  // Let the app keep running by returning an empty result.
    };
  }

  fetchDonationsIfNeeded(): Observable<DonationResponse> { // Return DonationResponse, not Donation[]
    return this.http.post<DonationResponse>(Endpoints.donations.getAll, {}).pipe(
      tap(response => {
        console.log('Fetched donations:', response);
      }),
      catchError(error => {
        console.error('Error fetching donation data:', error);
        return of({ total: 0, items: [] }); // Return empty DonationResponse in case of error
      })
    );
  }

  addMember(memberData: any): Observable<any> {
    console.log("Formatted member to send:", JSON.stringify(memberData));

    let missingFields = this.validateMemberData(memberData);
    if (missingFields.length > 0) {
      console.error('Missing required fields:', missingFields.join(', '));
      return of({ error: `Missing required fields: ${missingFields.join(', ')}` });
    }

    if (memberData.birthDate) {
      memberData.birthDate = this.formatDate(memberData.birthDate);
    }

    console.log("Formatted member to send:", memberData);
    return this.http.post(Endpoints.members.add, memberData, { observe: 'response' }).pipe(
      tap(response => {
        if (response.status === 200) {
          console.log('Member added successfully:', response.body);
        }
      }),
      catchError(error => {
        console.error('Error adding new member:', error);
        return of({ error: 'Error occurred while adding member.', details: error.message });
      })
    );
  }

  private validateMemberData(data: any): string[] {
    const requiredFields = ['firstName', 'lastName', 'phone', 'email', 'city', 'street', 'number', 'apartment', 'SMSInd', 'birthDate', 'parashatBarMitzva', 'totalFamilyMembers'];
    return requiredFields.filter(field => !data[field]);
  }

  private formatDate(date: any): string {
    const d = new Date(date);
    return `${d.getFullYear()}-${(d.getMonth() + 1).toString().padStart(2, '0')}-${d.getDate().toString().padStart(2, '0')}`;
  }

  openAddMemberDialog(): Observable<any> {
    const dialogRef = this.dialog.open(AddMemberFormComponent, { width: '250px' });

    return dialogRef.afterClosed().pipe(
      tap(result => {
        if (result) {
          this.memberCache.next([]); // Invalidate cache if a member is added
          this.fetchMembersIfNeeded().subscribe(); // Re-fetch members list
        }
      })
    );
  }

  searchMemberByName(name: string): Observable<any> {
    return this.http.post<any>(`${Endpoints.members.details}/searchMember`, { name });
  }

  getMemberById(id: number): Observable<any> {
    return this.memberCache.pipe(
      map(members => members.find(member => member.id === id))
    );
  }

  getMemberData(): Observable<any[]> {
    return this.memberCache.asObservable();
  }

  getDonationData(): Observable<any[]> {
    return this.donateCache.asObservable();
  }
}

export interface MemberResponse {
  id: number;
  firstName: string;
  lastName: string;
  phone: string;
  email: string;
  city: string;
  street: string;
  number: string;
  apartment: string;
  smsInd: number;
  birthDate: string;
  parashatBarMitzva: string;
  totalFamilyMembers: number;
  synagogue?: string;
}

interface DonationResponse {
  total: number;
  items: Donation[];
}

  datePledged: string;
  donatedFor: string;
  donorName: string;
  id: number;
  paid: number;

  parashaName: string;
  paymentDate: string | null;
  pledgeAmount: number;
  remarks: string | null;
  total_expenses: number;
  total_paid: number;
  datePledgedConverted?: Date;  // Optional Date property to store the converted date
}
