import { Component, EventEmitter, Input, Output } from '@angular/core'; 

@Component({
  selector: 'app-pagination',
  templateUrl: './pagination.component.html',
  styleUrls: ['./pagination.component.scss']
})
export class PaginationComponent {
  @Input() totalItems: number = 0;
  @Input() currentPage: number = 1;
  @Input() itemsPerPage: number = 10;
  @Output() pageChanged: EventEmitter<number> = new EventEmitter<number>();
  @Output() itemsPerPageChanged: EventEmitter<number> = new EventEmitter<number>();

  itemsPerPageOptions: number[] = [10, 20, 50, 100, 1000];

  get totalPages(): number {
    return this.totalItems > 0 ? Math.ceil(this.totalItems / this.itemsPerPage) : 1;
  }

  goToPage(page: number): void {
    if (page >= 1 && page <= this.totalPages && page !== this.currentPage) {
      this.currentPage = page;
      this.pageChanged.emit(this.currentPage);  // Emit the page number directly
    }
  }

  goToPreviousPage(): void {
    if (this.currentPage > 1) {
      this.goToPage(this.currentPage - 1);
    }
  }

  goToNextPage(): void {
    if (this.currentPage < this.totalPages) {
      this.goToPage(this.currentPage + 1);
    }
  }

  getPageRange(): number[] {
    const rangeSize = 2;
    const start = Math.max(1, this.currentPage - rangeSize);
    const end = Math.min(this.totalPages, this.currentPage + rangeSize);
    const pages: number[] = [];
    for (let i = start; i <= end; i++) {
      pages.push(i);
    }
    return pages;
  }

  onItemsPerPageChange(event: any): void {
    const newItemsPerPage = Number(event.target.value);
    if (newItemsPerPage !== this.itemsPerPage) {
      this.itemsPerPage = newItemsPerPage;
      this.itemsPerPageChanged.emit(newItemsPerPage);
      this.goToPage(1);  // Reset to the first page when items per page change
    }
  }
}

<div class="pagination-container">
    <!-- Dropdown for selecting items per page -->
    <label for="itemsPerPage">Items per page:</label>
    <select id="itemsPerPage" [(ngModel)]="itemsPerPage" (change)="onItemsPerPageChange($event)">
      <option *ngFor="let option of itemsPerPageOptions" [value]="option">{{ option }}</option>
    </select>
  
    <!-- Pagination buttons -->
    <nav *ngIf="totalItems > itemsPerPage">
      <ul class="pagination justify-content-center">
        <li class="page-item" [class.disabled]="currentPage === 1">
          <button class="page-link" (click)="goToPreviousPage()">Previous</button>
        </li>
  
        <!-- Display two pages before the current page -->
        <li *ngFor="let page of getPageRange()" class="page-item" [class.active]="page === currentPage">
          <button class="page-link" (click)="goToPage(page)">{{ page }}</button>
        </li>
  
        <li class="page-item" [class.disabled]="currentPage === totalPages">
          <button class="page-link" (click)="goToNextPage()">Next</button>
        </li>
      </ul>
    </nav>
</div>
