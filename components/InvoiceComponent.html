import { Component } from '@angular/core';
import { InvoiceResponse } from '../models/invoice-response.model';
import { InvoiceService } from 'app/invoice.service';
import { InvoiceRequest, InvoiceItem, Payment } from 'app/models/invoice-request.model';

@Component({
  selector: 'app-invoice',
  templateUrl: './invoice.component.html',
  styleUrls: ['./invoice.component.scss']
})
export class InvoiceComponent {

  invoiceRequest: InvoiceRequest = {
    type: 400,
    date: new Date().toISOString().split('T')[0], // Format date as YYYY-MM-DD
    vatType: 1,
    lang: 'he',
    currency: 'ILS',
    description: '',
    remarks: '',
    footer: '',
    emailContent: '',
    client: {
      self: false,
      name: '',
      emails: []
    },
    rounding: true,
    signed: true,
    income: [],
    payment: []
  };

  newItem: InvoiceItem = {
    catalogNum: '',
    description: '',
    quantity: 1,
    price: 0,
    currency: 'ILS',
    currencyRate: 1,
    vatRate: 0,
    vatType: 2
  };

  newPayment: Payment = {
    type: 1,
    price: 0,
    currency: 'ILS',
    currencyRate: 1,
    date: new Date().toISOString().split('T')[0]
  };

  constructor(private invoiceService: InvoiceService) {}

  addItem() {
    this.invoiceRequest.income.push({ ...this.newItem });
    this.newItem = {
      catalogNum: '',
      description: '',
      quantity: 1,
      price: 0,
      currency: 'ILS',
      currencyRate: 1,
      vatRate: 0,
      vatType: 2
    };
  }

  addPayment() {
    this.invoiceRequest.payment.push({ ...this.newPayment });
    this.newPayment = {
      type: 1,
      price: 0,
      currency: 'ILS',
      currencyRate: 1,
      date: new Date().toISOString().split('T')[0]
    };
  }

  createInvoice() {
    this.invoiceService.createInvoice(this.invoiceRequest).subscribe(
      (response: InvoiceResponse) => {
        console.log('Invoice created successfully', response);
      },
      (error) => {
        console.error('Error creating invoice', error);
      }
    );
  }
}
