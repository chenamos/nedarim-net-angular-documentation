  addDonation() {
    // Format date before submission
    if (this.newRow.datePledged) {
      const inputDate = new Date(this.newRow.datePledged);
      if (!isNaN(inputDate.getTime())) {
        const formattedDate = this.formatDate(inputDate);
        this.newRow.datePledged = formattedDate;
      } else {
        alert('Invalid date provided.');
        return;
      }
    }

    const donationWithYear = {
      ...this.newRow,
      parashaName: `${this.newRow.parashaName} ${this.newRow.year}`
    };

    this.dataService.addNewRecord(donationWithYear).subscribe({
      next: (res) => {
        alert(res.message || 'Donation added successfully!');
        this.resetForm();
        this.donationAdded.emit(true);  // Emit success event
      },
      error: () => {
        alert('Error occurred while adding donation.');
        this.donationAdded.emit(false);  // Emit failure event
      }
    });
  }

  formatDate(date: Date): string {
    const day = ('0' + date.getDate()).slice(-2);
    const month = ('0' + (date.getMonth() + 1)).slice(-2);
    const year = date.getFullYear();
    return `${year}-${month}-${day}`;
  }

  resetForm() {
    this.newRow = {
      donorName: '',
      donatedFor: this.donatedForOptions[0],
      year: this.years[0],
      parashaName: this.parashot[0],
      pledgeAmount: null,
      datePledged: '',
      remarks: ''
    };
  }
