import { Component, EventEmitter, Output } from '@angular/core';
import { FormControl, NgForm } from '@angular/forms';
import { MemberService } from 'app/member.service';
import { Observable } from 'rxjs';
import { startWith, map } from 'rxjs/operators';

@Component({
  selector: 'app-add-member-form',
  templateUrl: './add-member-form.component.html',
  styleUrls: ['./add-member-form.component.scss']
})
export class AddMemberFormComponent {
  @Output() memberAdded = new EventEmitter<boolean>();
  
  member = {
    firstName: '',
    lastName: '',
    phone: '',
    email: '',
    city: '',
    street: '',
    number: '',
    apartment: '',
    SMSInd: 0,
    birthDate: new Date(),
    parashatBarMitzva: '',
    totalFamilyMembers: 0
  };

  // List of all Parashot
  parashot = [
    'בראשית', 'נח', 'לך לך', 'וירא', 'חיי שרה', 'תולדות', 'ויצא', 'וישלח', 'וישב', 'מקץ', 'ויגש', 'ויחי',
    'שמות', 'וארא', 'בא', 'בשלח', 'יתרו', 'משפטים', 'תרומה', 'תצוה', 'כי תשא', 'ויקהל', 'פקודי', 'ויקהל-פקודי',
    'ויקרא', 'צו', 'שמיני', 'תזריע', 'מצורע', 'תזריע-מצורע', 'אחרי מות', 'קדושים', 'אחרי מות-קדושים', 'אמור', 'בהר', 'בחוקותי', 'בהר-בחוקותי',
    'במדבר', 'נשא', 'בהעלותך', 'שלח', 'קורח', 'חוקת', 'בלק', 'פינחס', 'מטות', 'מסעי', 'מטות-מסעי',
    'דברים', 'ואתחנן', 'עקב', 'ראה', 'שופטים', 'כי תצא', 'כי תבוא', 'ניצבים', 'וילך', 'ניצבים-וילך', 'האזינו', 'וזאת הברכה'
  ];

  parashatControl = new FormControl(); // Control for Parashat Bar Mitzvah select field
  filteredParashot: Observable<string[]>; // Observable for filtered Parashot

  constructor(private memberService: MemberService) {
    this.filteredParashot = this.parashatControl.valueChanges.pipe(
      startWith(''),
      map(value => {
        const filtered = this.filterParashot(value);
        console.log("Filtering Parashot with:", value, "Result:", filtered);
        return filtered;
      })
    );
    console.log("AddMemberFormComponent initialized.");
  }

  private filterParashot(value: string): string[] {
    const filterValue = value.toLowerCase();
    return this.parashot.filter(option => option.toLowerCase().includes(filterValue));
  }

  onSubmitMemberForm(form: NgForm): void {
    console.log("Submitting form with data:", form.value);
    this.addMember(form);
  }
 

  addMember(form: NgForm): void {
    console.log("Attempting to add member:", this.member);
    if (form.valid) {
      this.memberService.addMember(this.member).subscribe({
        next: () => {
          console.log('Member added successfully');
          this.memberAdded.emit(true);
          form.resetForm(); // Reset the form after successful submission
        },
        error: error => {
          console.error('Error adding member:', error);
          this.memberAdded.emit(false);
        }
      });
    } else {
      console.log("Form is invalid, please review the entries");
      alert('Please fill all required fields.');
    }
  }

  resetForm(form: NgForm): void {
    console.log("Resetting form");
    form.resetForm(); // Reset the form
    this.member = {
      firstName: '',
      lastName: '',
      phone: '',
      email: '',
      city: '',
      street: '',
      number: '',
      apartment: '',
      SMSInd: 0,
      birthDate: new Date(),
      parashatBarMitzva: '', // Reset this field
      totalFamilyMembers: 0
    };
  }
}
