import { Component, OnInit } from '@angular/core';
import { FormControl } from '@angular/forms';
import { MemberService, MemberResponse, Donation } from 'app/member.service';

@Component({
  selector: 'app-member-account-page',
  templateUrl: './member-account-page.component.html',
  styleUrls: ['./member-account-page.component.scss']
})
export class MemberAccountPageComponent implements OnInit {
  members: MemberResponse[] = [];
  donations: Donation[] = []; // Updated to match Donation type
  selectedDonorDonations: any[] = [];
  selectedDonorId: number | null = null;
  selectedMember: MemberResponse | null = null;
  startDate: string = '';
  endDate: string = '';
  net_balance: number = 0;
  totalPledged: number = 0;
  totalPaid: number = 0;
  mostFrequentDonatedFor: string = '';
  durationWithUs: string = '';
  highestPledgeAmount: number = 0; // Variable to store the highest pledge amount
  distinctDonors: string[] = []; // Array to store distinct donor names
  searchText: string = ''; // Search text for filtering donors
  totalExpenses: number = 0;
  filteredMembers: MemberResponse[] = []; // Add filteredMembers property
  donorFilterCtrl = new FormControl();
  donorSelectCtrl = new FormControl(); // This controls the actual selected value
  constructor(private memberService: MemberService) {}

  ngOnInit(): void {
    console.log("Initializing MemberAccountPageComponent...");
    this.fetchAllData();

    // Subscribe to changes in the donorFilterCtrl to filter the member list dynamically
    this.donorFilterCtrl.valueChanges.subscribe(val => {
      this.filterDonors(val);
    });
  }
  
  testEvent(): void {
    console.log('Event triggered');
  }
  
  logDropdownOpen(isOpen: boolean): void {
    if (isOpen) {
      console.log('Current filtered members:', this.filteredMembers);
    }
  }
  
  fetchAllData(): void {
    console.log("Fetching all data for members and donations...");
  
    // Fetch members data
    this.memberService.fetchMembersIfNeeded().subscribe(members => {
      console.log("Members fetched:", members); // Log fetched members
      this.members = members;
  
      if (members.length > 0) {
        this.selectedDonorId = this.members[0].id;
        console.log("Selected Donor ID after fetching members:", this.selectedDonorId);
        this.onDonorChange(this.selectedDonorId);
      }
  
      // Fetch donations data (no 'items' field in response)
      this.memberService.fetchDonationsIfNeeded().subscribe(response => {
        console.log("Raw donations data from server:", response); // Log the raw server response
  
        // Directly assign the response as donations since there's no 'items' field
        this.donations = Array.isArray(response) ? response : [];
        console.log("Donations array after assignment:", this.donations);
  
        if (this.selectedDonorId) {
          this.filterDonations(); // Filter based on selected donor
        }
  
        this.populateDistinctDonors(); // Populate donor names for filtering
      }, error => {
        console.error("Error fetching donations:", error);
      });
    }, error => {
      console.error("Error fetching members:", error);
    });
  }
  
  
  populateDistinctDonors(): void {
    this.distinctDonors = this.members.map(member => `${member.firstName} ${member.lastName}`);
    console.log("Distinct donors populated:", this.distinctDonors);
  }

  filterDonations(): void {
    console.log("Filtering donations, Selected Donor ID:", this.selectedDonorId);
  
    if (this.selectedDonorId !== null) {
      const selectedDonor = this.members.find(member => member.id === Number(this.selectedDonorId));
      console.log("Selected Donor found:", selectedDonor);
  
      if (selectedDonor) {
        const donorName = `${selectedDonor.firstName} ${selectedDonor.lastName}`.trim().toLowerCase();
        console.log("Filtering donations for donor:", donorName);
  
        console.log("All donations before filtering:", this.donations); // Log all donations before filtering
  
        // Ensure donations is not empty and filter donations by donor name
        this.selectedDonorDonations = this.donations
          .filter(donation => {
            // Null check for donorName in the donation object
            if (!donation.donorName) {
              console.warn("Skipping donation with null donorName:", donation);
              return false; // Skip this donation if donorName is null or undefined
            }
  
            const donationName = donation.donorName.trim().toLowerCase(); // Trim and lowercase for consistency
            return donationName === donorName;
          })
          .sort((a, b) => new Date(b.datePledged).getTime() - new Date(a.datePledged).getTime()); // Sort by date
  
        console.log("Filtered Donor Donations:", this.selectedDonorDonations); // Log filtered donations
  
        if (this.selectedDonorDonations.length > 0) {
          console.log("Donations found for donor:", donorName);
          this.calculateDonorTotals(); // Calculate totals if donations found
          this.selectedMember = selectedDonor; // Set the selected member
        } else {
          console.log("No donations found for donor:", donorName);
        }
      } else {
        console.log("No donor found for selected ID:", this.selectedDonorId);
      }
    } else {
      console.log("Selected Donor ID is null, skipping filtering.");
    }
  }
  
  
  calculateDonorTotals(): void {
    this.totalPledged = this.selectedDonorDonations.reduce((total, donation) => total + donation.pledgeAmount, 0);
    this.totalPaid = this.selectedDonorDonations.reduce((total, donation) => total + donation.paid, 0);
    console.log("Total Pledged:", this.totalPledged, "Total Paid:", this.totalPaid);

    // Access currentAmount from the first donation in the array
    this.net_balance = this.selectedDonorDonations.length > 0 ? this.selectedDonorDonations[0].currentAmount : 0;
    this.calculateMostFrequentDonatedFor();
    this.calculateDurationWithUs();
    this.calculateHighestPledgeAmount();
  }

  calculateMostFrequentDonatedFor(): void {
    const frequency = {};
    let maxFreq = 0;
    this.selectedDonorDonations.forEach(donation => {
      if (donation.donatedFor) {
        frequency[donation.donatedFor] = (frequency[donation.donatedFor] || 0) + 1;
        if (frequency[donation.donatedFor] > maxFreq) {
          maxFreq = frequency[donation.donatedFor];
          this.mostFrequentDonatedFor = donation.donatedFor;
        }
      }
    });
    this.mostFrequentDonatedFor = maxFreq === 0 ? 'אין תרומות' : this.mostFrequentDonatedFor;
    console.log("Most frequent donated for:", this.mostFrequentDonatedFor);
  }

  calculateDurationWithUs(): void {
    const validDates = this.selectedDonorDonations
      .filter(donation => donation.datePledged)
      .map(donation => new Date(donation.datePledged).getTime());

    if (validDates.length > 0) {
      const earliestDate = new Date(Math.min(...validDates));
      const currentDate = new Date();
      const diff = currentDate.getTime() - earliestDate.getTime();
      const years = Math.floor(diff / (1000 * 3600 * 24 * 365));
      const months = Math.floor((diff % (1000 * 3600 * 24 * 365)) / (1000 * 3600 * 24 * 30));
      this.durationWithUs = `${years} שנים (${months} חודשים)`;
    } else {
      this.durationWithUs = 'לא ניתן לחשב משך זמן';
    }
    console.log("Duration with us:", this.durationWithUs);
  }

  calculateHighestPledgeAmount(): void {
    this.highestPledgeAmount = this.selectedDonorDonations.length > 0
      ? Math.max(...this.selectedDonorDonations.map(donation => donation.pledgeAmount || 0))
      : 0;
    console.log("Highest Pledge Amount:", this.highestPledgeAmount);
  }

  
  
  updateDateRange(): void {
    console.log("Updating date range:", this.startDate, this.endDate);
    // Implement your logic here
  }

  filterDonors(val: string): void {
    console.log("Filtering donors with value:", val);
    this.filteredMembers = this.members.filter(member =>
      `${member.firstName} ${member.lastName}`.toLowerCase().includes(val.toLowerCase())
    );
    console.log("Filtered Members:", this.filteredMembers);
  }
  
  calculateTotalExpenses(): void {
    this.totalExpenses = this.selectedDonorDonations.reduce((total, donation) => total + donation.total_expenses, 0);
    console.log("Total Expenses:", this.totalExpenses);
  }
  
  fetchAllMembers(): void {
    this.memberService.getMemberData().subscribe(
      members => {
        console.log('Members fetched:', members);
        if (members.length > 0) {
          this.members = members;
          this.filteredMembers = [...members]; // Reassign to trigger Angular's change detection
          console.log('Filtered Members:', this.filteredMembers); // Debug the assignment
        } else {
          console.log('No members data returned from the server.');
        }
      },
      error => {
        console.error('Error fetching members:', error);
      }
    );
  }

  onDonorChange(selectedId: number): void {
    console.log("Donor selection changed, new ID:", selectedId);
    this.selectedDonorId = selectedId;
    this.selectedMember = this.members.find(member => member.id === selectedId);
    if (this.selectedMember) {
      console.log("Selected member details updated:", this.selectedMember);
      this.filterDonations();
    } else {
      console.error("No member found with ID:", selectedId);
    }
  }
}
