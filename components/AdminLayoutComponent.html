import { Component, OnInit, AfterViewInit } from '@angular/core';
import { Location, PopStateEvent } from '@angular/common';
import { Router, NavigationEnd, NavigationStart } from '@angular/router';
import { filter, Subscription } from 'rxjs';
import { AppConfigService } from '../../app-config.service';
import PerfectScrollbar from 'perfect-scrollbar';

@Component({
  selector: 'app-admin-layout',
  templateUrl: './admin-layout.component.html',
  styleUrls: ['./admin-layout.component.scss']
})
export class AdminLayoutComponent implements OnInit, AfterViewInit {
  appName: string;
  private _router: Subscription;
  private lastPoppedUrl: string;
  private yScrollStack: number[] = [];


  constructor(public location: Location, private router: Router, private configService: AppConfigService) {}

  ngOnInit() {
    this.appName = this.configService.getAppName(); // Get app name on component initialization
    const isWindows = navigator.platform.indexOf('Win') > -1;

    if (isWindows && !document.getElementsByTagName('body')[0].classList.contains('sidebar-mini')) {
      // If on Windows OS, activate PerfectScrollbar
      document.getElementsByTagName('body')[0].classList.add('perfect-scrollbar-on');
    } else {
      document.getElementsByTagName('body')[0].classList.remove('perfect-scrollbar-off');
    }

    const elemMainPanel = <HTMLElement>document.querySelector('.main-panel');
    const elemSidebar = <HTMLElement>document.querySelector('.sidebar .sidebar-wrapper');

    this.location.subscribe((ev: PopStateEvent) => {
      this.lastPoppedUrl = ev.url;
    });

    this.router.events.subscribe((event: any) => {
      if (event instanceof NavigationStart) {
        if (event.url !== this.lastPoppedUrl) {
          this.yScrollStack.push(window.scrollY);
        }
      } else if (event instanceof NavigationEnd) {
        if (event.url === this.lastPoppedUrl) {
          this.lastPoppedUrl = undefined;
          window.scrollTo(0, this.yScrollStack.pop());
        } else {
          window.scrollTo(0, 0);
        }
      }
    });

    this._router = this.router.events.pipe(filter(event => event instanceof NavigationEnd)).subscribe(() => {
      elemMainPanel.scrollTop = 0;
      elemSidebar.scrollTop = 0;
    });

    // Initialize PerfectScrollbar if conditions are met
    this.initializePerfectScrollbar(elemMainPanel);
    this.initializePerfectScrollbar(elemSidebar);

    // Scroll to top button functionality
    window.onscroll = () => {
        const scrollToTopBtn = document.getElementById('scrollToTopBtn');
        if (scrollToTopBtn) {
          if (document.body.scrollTop > 200 || document.documentElement.scrollTop > 200) {
            scrollToTopBtn.classList.add('show-scroll');
          } else {
            scrollToTopBtn.classList.remove('show-scroll');
          }
        }
      };
      
  }

  ngAfterViewInit() {
    this.runOnRouteChange();
  }

  isMaps(path: string) {
    const titlee = this.location.prepareExternalUrl(this.location.path());
    return path !== titlee;
  }

  runOnRouteChange(): void {
    const elemMainPanel = <HTMLElement>document.querySelector('.main-panel');
    
    if (elemMainPanel) {
      // Proceed only if the main panel exists
      this.initializePerfectScrollbar(elemMainPanel);
    } else {
      console.warn('Main panel element not found. Skipping PerfectScrollbar initialization.');
    }
  }

  initializePerfectScrollbar(element: HTMLElement) {
    if (element && window.matchMedia(`(min-width: 960px)`).matches && !this.isMac()) {
      try {
        if (!element.getAttribute('ps-initialized')) {
          const ps = new PerfectScrollbar(element);
          ps.update();
          element.setAttribute('ps-initialized', 'true');
        }
      } catch (error) {
        console.error('Error initializing PerfectScrollbar:', error);
      }
    }
  }
  
  isMac(): boolean {
    return navigator.platform.toUpperCase().indexOf('MAC') >= 0 || navigator.platform.toUpperCase().indexOf('IPAD') >= 0;
  }

scrollToTop() {
  const elemMainPanel = <HTMLElement>document.querySelector('.main-panel');
  if (elemMainPanel) {
    console.log('Scroll to top button clicked - scrolling main panel');
    elemMainPanel.scrollTo({ top: 0, behavior: 'smooth' });
  } else {
    console.log('Scroll to top button clicked - scrolling window');
    window.scrollTo({ top: 0, behavior: 'smooth' });
  }
}

}
