import { ChangeDetectorRef, Component } from '@angular/core';
import { Router } from '@angular/router';
import { DataService } from '../data.service';

@Component({
  selector: 'app-pledge-payment',
  templateUrl: './pledge-payment.component.html',
  styleUrls: ['./pledge-payment.component.scss']
})
export class PledgePaymentComponent {
  paymentData = {
    donorName: '',  
    paid: null,
    paid_to: '',
    payment_date: ''
  };

  constructor(private router: Router, private dataService: DataService, private cdRef: ChangeDetectorRef) { }

  processPayment() {
    console.log('Processing payment:', this.paymentData);
  
    this.dataService.addPaymentRecord(this.paymentData).subscribe({
      next: (response) => {
        // Log the payment response
        console.log('Payment response:', response);
  
        // Alert the user that payment was successful
        alert('Payment processed successfully!');
  
        // Reset the form after the payment is successful
        this.resetForm();
  
        // Navigate to a different route (donators-managing) after processing the payment
        this.router.navigate(['/donators-managing']); // Adjust the route as needed
      },
      error: (error) => {
        console.error('Error processing payment:', error);
        alert('Error occurred while processing payment.');
      }
    });
  }
  
  


  resetForm() {
    this.paymentData = {
      donorName: '',
      paid: null,
      paid_to: '',
      payment_date: ''
    };
  }
}
