import { Component, OnInit } from '@angular/core';
import { DataService } from '../data.service';
import { UserService } from 'app/user.service';

@Component({
  selector: 'app-expenditures-display',
  templateUrl: './expenditures-display.component.html',
  styleUrls: ['./expenditures-display.component.scss']
})
export class ExpendituresDisplayComponent implements OnInit {
  expenditures: any[] = [];
  total_expenses: number = 0;
  currentUser: any;  // Renaming to differentiate from AppUser field

  constructor(private dataService: DataService, private userService: UserService) {}

  ngOnInit(): void {
    this.dataService.getExpendituresData().subscribe(
      data => {
        // Sort the expenditures by expenditure_date in descending order
        this.expenditures = data.sort((a, b) => {
          const dateA = new Date(a.expenditure_date).getTime();
          const dateB = new Date(b.expenditure_date).getTime();
          return dateB - dateA;  // Sort descending by date
        });
      },
      error => {
        console.error('Error fetching expenditures data:', error);
      }
    );

    this.userService.getCurrentUserInfo().subscribe(
      (userInfo) => {
        this.currentUser = userInfo;
        console.log('Current user info:', this.currentUser);
      },
      (error) => console.error('Error fetching current user info:', error)
    );
  }
}
