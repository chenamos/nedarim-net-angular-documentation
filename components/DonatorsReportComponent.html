import { Component, OnInit } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { MatDialog } from '@angular/material/dialog';
import { DonatorReportSelModalComponent } from 'app/donator-report-sel-modal/donator-report-sel-modal.component';
import { Router } from '@angular/router';
import { ReportService } from 'app/report.service';
import { Endpoints } from 'app/constants/base-url.constants';

@Component({
  selector: 'app-donators-report',
  templateUrl: './donators-report.component.html',
  styleUrls: ['./donators-report.component.scss']
})
export class DonatorsReportComponent implements OnInit {
  reportData: any[] = [];

  constructor(
    private http: HttpClient, 
    public dialog: MatDialog, 
    private router: Router,
    private reportService: ReportService
  ) {
    console.log('DonatorsReportComponent constructed');
  }

  ngOnInit(): void {
    console.log('DonatorsReportComponent: ngOnInit started');
    try {
      this.fetchReportData();
      console.log('DonatorsReportComponent: Data fetch initiated');
    } catch (error) {
      console.error('DonatorsReportComponent: Error during data fetch', error);
    }
    console.log('DonatorsReportComponent: ngOnInit completed');
  }
  
  fetchReportData(): void {
    console.log('Fetching data from server...');
    this.http.get<any[]>(Endpoints.reports.filteredReport)  // Use the centralized endpoint here
      .subscribe({
        next: (data) => {
          console.log('Data fetched successfully:', data);
          this.reportData = data;
        },
        error: (error) => console.error('Failed to fetch data:', error)
      });
  }

  openFilterDialog(): void {
    console.log('Opening filter dialog...');
    const dialogRef = this.dialog.open(DonatorReportSelModalComponent, {
      width: '250px',
      data: {}
    });
  
    dialogRef.componentInstance.applyFilter.subscribe(filterData => {
      console.log('Filter data received:', filterData);
      if (filterData) {
        this.fetchReportDataFiltered(filterData);
      }
    });
  
    dialogRef.afterClosed().subscribe(() => {
      console.log('Filter dialog has been closed');
    });
  }
  

  fetchReportDataFiltered(filterData: any): void {
    console.log('Applying filter to data:', filterData);
    this.reportService.getFilteredReports(filterData).subscribe({
      next: (filteredData) => {
        console.log('Filtered data received in parent component:', filteredData);
        this.reportService.setFilteredData(filteredData);
        this.router.navigate(['/donators-managing']);
      },
      error: (error) => console.error('Failed to fetch filtered data:', error)
    });
  }

  printReport(): void {
    // Clone the print section HTML
    const printContents = document.querySelector('.print-section').innerHTML;
  
    // Create a new window to print
    const popupWindow = window.open('', '_blank', 'width=800,height=600');
    
    // Write the cloned HTML content into the new window's document
    popupWindow.document.open();
    popupWindow.document.write(`
      <html>
        <head>
          <title>Print Report</title>
          <style>
            body { font-family: Arial, sans-serif; direction: rtl; }
            table { width: 100%; border-collapse: collapse; margin-bottom: 10px; table-layout: fixed; }
            table, th, td { border: 1px solid black; padding: 8px; text-align: center; word-wrap: break-word; }
            th { background-color: #f2f2f2; }
  
            .rtl-table th, .rtl-table td {
              padding: 8px;
              text-align: center;
            }
  
            .rtl-table {
              width: 100%;
              table-layout: fixed;
            }
  
            .rtl-table tbody tr, .rtl-table thead tr {
              display: table-row;
              direction: rtl;
            }
          </style>
        </head>
        <body onload="window.print(); window.close();">
          ${printContents}
        </body>
      `);
    popupWindow.document.close();
  }
  
  
}
