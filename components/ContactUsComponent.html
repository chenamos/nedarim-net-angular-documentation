import { Component } from '@angular/core';
import { NgForm } from '@angular/forms';
import { ContactUsService } from 'app/contact-us.service';

@Component({
  selector: 'app-contact-us',
  templateUrl: './contact-us.component.html',
  styleUrls: ['./contact-us.component.scss']
})
export class ContactUsComponent {
  contactDetails = {
    donor_name: '',
    email: '',
    message: ''
  };

  constructor(private contactUsService: ContactUsService) {}

  // Method to submit the contact form
  submitContactForm(form: NgForm): void {
    if (form.valid) {
      console.log('Sending contact form data:', this.contactDetails);
      this.contactUsService.sendContactData(this.contactDetails).subscribe({
        next: (response) => {
          console.log('Message sent successfully:', response);
          alert('Your message has been sent successfully!');  // <-- Show success alert
          this.resetForm(form);
        },
        error: (error) => {
          console.error('Error sending message:', error);
          alert('There was an error sending your message. Please try again later.');  // <-- Show error alert
        }
      });
    } else {
      alert('Please fill out all required fields.');
    }
  }

  // Method to reset the form after successful submission or when the user chooses to reset the form
  resetForm(form: NgForm): void {
    form.resetForm();
    this.contactDetails = {
      donor_name: '',
      email: '',
      message: ''
    };
  }
}
