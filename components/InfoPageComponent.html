import { Component, OnInit, AfterViewInit, Renderer2, ElementRef } from '@angular/core';
import { ZmanimService } from '../zmanim.service';
import { AppConfigService } from 'app/app-config.service';

@Component({
  selector: 'app-info-page',
  templateUrl: './info-page.component.html',
  styleUrls: ['./info-page.component.scss']
})
export class InfoPageComponent implements OnInit, AfterViewInit {
  public zmanim: any;
  public halachaOfTheDay: string;
  public mishnaOfTheDay: string;
  public appName: string;

  constructor(
    private zmanimService: ZmanimService,
    private configService: AppConfigService,
    private renderer: Renderer2,
    private el: ElementRef
  ) {}

  ngOnInit(): void {
    this.appName = this.configService.getAppName(); // Get app name on component initialization
    this.fetchDailyInfo();
    // Refresh every 24 hours
    setInterval(() => {
      this.fetchDailyInfo();
    }, 86400000); // 24 hours
  }

  fetchDailyInfo() {
    // Fetch zmanim
    this.zmanimService.getZmanim(31.7683, 35.2137, "Asia/Jerusalem").subscribe(data => {
      this.zmanim = data;
    });

    // Fetch Halacha and Mishna
    this.halachaOfTheDay = this.zmanimService.getDailyHalacha();
    this.mishnaOfTheDay = this.zmanimService.getDailyMishna();
  }

  ngAfterViewInit() {
    const button = this.el.nativeElement.querySelector('#fullscreen-button');
    if (button) {
      button.addEventListener('click', () => this.toggleFullScreen());
    }
  }

  toggleFullScreen(): void {
    const infoContainer = this.el.nativeElement.querySelector('.info-container');
    if (!document.fullscreenElement) {
      infoContainer.requestFullscreen().catch(err => {
        console.error(`Error attempting to enable full-screen mode: ${err.message} (${err.name})`);
      });
      this.renderer.addClass(document.body, 'fullscreen');
    } else {
      if (document.exitFullscreen) {
        document.exitFullscreen();
      }
      this.renderer.removeClass(document.body, 'fullscreen');
    }
  }
}
