import { Component, EventEmitter, OnInit, Output } from '@angular/core';
import { MatDialogRef } from '@angular/material/dialog';
import { ReportService } from '../report.service';

@Component({
  selector: 'app-donator-report-sel-modal',
  templateUrl: './donator-report-sel-modal.component.html',
  styleUrls: ['./donator-report-sel-modal.component.scss']
})
export class DonatorReportSelModalComponent implements OnInit {
  data: any = {};
  selectedReportType: string = '';
  donorNames: string[] = [];
  parshaNames: string[] = [];
  donatedFor: string[] = [];

  @Output() applyFilter = new EventEmitter<any>(); 

  constructor(
    public dialogRef: MatDialogRef<DonatorReportSelModalComponent>,
    private reportService: ReportService
  ) {}

  ngOnInit() {
    this.reportService.getDonorNames().subscribe(names => {
      this.donorNames = names;
    });
    this.reportService.getDonatedFor().subscribe(donatedFor => {
      this.donatedFor = donatedFor;
    });
    this.reportService.getParshaNames().subscribe(parshaNames => {
      this.parshaNames = parshaNames;
    });
  }

  onClose(): void {
    this.dialogRef.close();
  }

  onApplyFilter(): void {
    const filterData: any = {};
  
    if (this.data.donorName?.trim()) {
      filterData['donorName'] = this.data.donorName.trim();
    }
    if (this.data.donatedFor?.trim()) {
      filterData['donatedFor'] = this.data.donatedFor.trim();
    }
    if (this.data.parashaName?.trim()) {
      filterData['parashaName'] = this.data.parashaName.trim();
    }
  
    console.log("Filter data to be applied:", filterData); // Log filterData before filtering
  
    this.reportService.getFilteredReports(filterData).subscribe(filteredData => {
      if (filteredData.length) {
        this.reportService.setFilteredData(filteredData);
        this.applyFilter.emit(filterData);  // Emit the filter data
        this.dialogRef.close();  // Close the modal after successful filtering
      } else {
        alert("No data matches the filter.");
      }
    }, error => {
      console.error("Error while filtering data:", error);
    });
  }
  
}
