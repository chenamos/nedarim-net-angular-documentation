import { ChangeDetectorRef, Component, OnInit } from '@angular/core';
import { DataService } from '../data.service';
import { ReportService } from '../report.service';
import { UserService } from 'app/user.service';
import { AppUser } from 'app/models/app-user.model';
import { HttpClient } from '@angular/common/http';


@Component({
  selector: 'app-donators-managing',
  templateUrl: './donators-managing.component.html',
  styleUrls: ['./donators-managing.component.scss']
})
export class DonatorsManagingComponent implements OnInit {
  data: any[] = [];
  newRow: any = {};
  originalData: any = {};
  total_paid: number = 0;
  total_expenses: number = 0;
  net_balance: number = 0;
  filteredData: any[] = [];
  currentUser: AppUser;
  currentUserInfo: any;
  paymentData: any = {};
  totalPledgeAmount: number = 0;  // Sum of selected pledge amounts
  isPaymentModalOpen: boolean = false;  // Flag for modal control
  selectedPledgeIds: any[] = [];
  selectedPledges: { id: number, amount: number }[] = [];
  showAddDonationModal: boolean = false;

  // Pagination properties
  currentPage: number = 1;
  itemsPerPage: number = 50;
  totalItems: number = 0;
  itemsPerPageOptions: number[] = [10, 20, 50, 100, 1000];  // Dropdown options for items per page
  isLoading: boolean = false;  // Prevent multiple simultaneous API calls
  showModal: boolean = false;

  constructor(
    private dataService: DataService,
    private userService: UserService,
    private reportService: ReportService,
    private changeDetectorRef: ChangeDetectorRef,
    private http: HttpClient
  ) {}

  ngOnInit(): void {
    console.log('DonatorsManagingComponent initialized');
    this.userService.getCurrentUserInfo().subscribe((userInfo) => {
      this.currentUser = userInfo;
      this.paymentData.terminalNumber = this.currentUser.terminalNumber;
      this.paymentData.paymentPassword = this.currentUser.paymentPassword;
    });

    this.reportService.getFilteredData().subscribe(filteredData => {
      if (filteredData && filteredData.length > 0) {
        this.data = filteredData;
        console.log('Filtered data received:', this.data);
      } else {
        this.loadPageData(this.currentPage);
      }
    });

    this.loadFinancialSummary();  // Load financial summary data
  }

  loadPageData(page: number): void {
    this.isLoading = true;
    this.dataService.getPaginatedData(page - 1, this.itemsPerPage).subscribe(
      (response: { items: any[]; total: number }) => {
        this.data = response.items || [];
        this.totalItems = response.total || 0;
        this.isLoading = false;
      },
      (error) => {
        console.error('Error fetching paginated data:', error);
        this.isLoading = false;
      }
    );
  }

  loadFinancialSummary(): void {
    this.dataService.getFinancialSummary().subscribe(
      (response) => {
        this.total_paid = response.total_paid || 0;
        this.total_expenses = response.total_expenses || 0;
        this.net_balance = this.total_paid - this.total_expenses;
      },
      (error) => {
        console.error('Error fetching financial summary:', error);
      }
    );
  }


  // Handle page change event correctly now
  onPageChange(page: number): void {
    if (page >= 1) {
      this.currentPage = page;
      this.loadPageData(this.currentPage);
    } else {
      console.warn('Invalid page number:', page);
    }
  }

  processData(response: any[]): void {
    console.log('processData called, response:', response);
    if (!response || response.length === 0) {
      console.error('Received an empty or undefined response:', response);
      return;
    }

    this.data = response.map(item => ({
      id: item.id,
      pledgeDate: item.datePledged,
      donorName: item.donorName,
      donatedFor: item.donatedFor,
      pledgeAmount: item.pledgeAmount,
      parashaName: item.parashaName || '',
      notes: item.remarks || '',
      paid: item.paid,
      paidTo: item.paidTo || '',
      paymentDate: item.paymentDate || '',
      currentAmount: item.currentAmount,
      selected: false
    }));

    if (response[0]) {
      this.total_paid = response[0].total_paid || 0;
      this.total_expenses = response[0].total_expenses || 0;
      this.net_balance = this.total_paid - this.total_expenses;
    }

    this.changeDetectorRef.detectChanges();
  }

  onItemsPerPageChange(itemsPerPage: number): void {
    if (itemsPerPage !== this.itemsPerPage) {
      this.itemsPerPage = itemsPerPage;
      this.currentPage = 1;  // Reset to first page on items per page change
      this.loadPageData(0);  // Zero-based for backend
    }
  }

  onPageSizeChange(): void {
    this.currentPage = 1; // Reset to page 1 when the page size changes
    this.loadPageData(this.currentPage); // Load data with new page size and page 1
  }

  addDonation() {
    console.log('Attempting to add donation:', this.newRow);
    let missingFields = [];
    if (!this.newRow.donorName) { missingFields.push('donorName'); }
    if (!this.newRow.pledgeAmount) { missingFields.push('pledgeAmount'); }
    if (missingFields.length > 0) {
      console.log('Missing required fields:', missingFields.join(', '));
      alert(`Please fill in all required fields: ${missingFields.join(', ')}.`);
    } else {
      console.log('All required fields are present');
      this.dataService.addNewRecord(this.newRow).subscribe({
        next: (response) => {
          alert('Donation record added successfully!');
          this.data.push(this.newRow);
          this.newRow = {};
          this.loadFinancialSummary();  // Update financial summary
          console.log('Updated data array:', this.data);
        },
        error: (error) => {
          console.error('Failed to add donation record:', error);
          alert('Failed to add record. Check console for errors.');
        }
      });
    }
  }

  printReport(): void {
    const printContents = document.querySelector('.print-section').innerHTML;
    const popupWindow = window.open('', '_blank', 'width=800,height=600');
    
    popupWindow.document.open();
    popupWindow.document.write(`
      <html>
        <head>
          <title>Print Report</title>
          <style>
            body { font-family: Arial, sans-serif; direction: rtl; }
            table { width: 100%; border-collapse: collapse; margin-bottom: 20px; table-layout: fixed; }
            table, th, td { border: 1px solid black; padding: 8px; text-align: center; word-wrap: break-word; }
            th { background-color: #f2f2f2; }
          </style>
        </head>
        <body onload="window.print(); window.close();">
          ${printContents}
        </body>
      </html>
    `);
    popupWindow.document.close();
  }

  formatAmount(amount: number, currencyCode: string): string {
    if (!amount && amount !== 0) {
      return ''; // Handle undefined or null values
    }

    const hasDecimals = amount % 1 !== 0;

    if (hasDecimals) {
      return new Intl.NumberFormat('he-IL', {
        style: 'currency',
        currency: currencyCode,
        minimumFractionDigits: 1,
        maximumFractionDigits: 1,
        currencyDisplay: 'narrowSymbol'
      }).format(amount);
    } else {
      return new Intl.NumberFormat('he-IL', {
        style: 'currency',
        currency: currencyCode,
        minimumFractionDigits: 0,
        maximumFractionDigits: 0,
        currencyDisplay: 'narrowSymbol'
      }).format(amount);
    }
  }

  openPaymentModal() {
    const selectedPledges = [];

    this.totalPledgeAmount = this.data
      .filter(item => item.selected)
      .reduce((total, item) => {
        if (item.selected) {
          selectedPledges.push({ id: item.id, amount: item.pledgeAmount });
        }
        return total + item.pledgeAmount;
      }, 0);

    if (this.totalPledgeAmount > 0) {
      this.isPaymentModalOpen = true;
      this.selectedPledges = selectedPledges;
    } else {
      alert('Please select at least one record to pay.');
    }
  }

  submitPayment() {
    console.log('Payment Data before submission:', this.paymentData);  // Log paymentData here
  
    if (!this.paymentData.cardNumber || !this.paymentData.cvv || !this.paymentData.expDate) {
      alert('Please provide complete payment details.');
      return;
    }
  
    if (this.totalPledgeAmount <= 0) {
      alert('No pledge amount selected for payment.');
      return;
    }
  
    const formattedExpDate = this.formatExpirationDate(this.paymentData.expDate);
  
    const payload = {
      cardNumber: this.paymentData.cardNumber,
      cvv: this.paymentData.cvv,
      expDate: formattedExpDate,
      transactionSum: this.totalPledgeAmount,
      pledges: this.selectedPledges
    };
  
    console.log('Formatted expiration date:', formattedExpDate);
    console.log('Payment request to be sent to the backend:', payload);  // Log payload here
  
    this.dataService.processPayment(payload).subscribe({
      next: (response) => {
        alert('Payment processed successfully!');
        this.clearPaymentData();
        this.closePaymentModal();
        this.loadFinancialSummary();
      },
      error: (error) => {
        if (error.status === 400) {
          alert('Bad request: Please check the payment details.');
        } else {
          alert(`Payment processing failed: ${error.message}`);
        }
      }
    });
  }
  
  

  formatExpirationDate(date: string): string {
    if (!date || date.indexOf('/') === -1) {
      return date;
    }

    const [month, year] = date.split('/');
    return `${month}${year.slice(-2)}`;
  }

  closePaymentModal() {
    this.isPaymentModalOpen = false;
  }

  clearPaymentData() {
    this.paymentData = {
      cardNumber: '',
      cvv: '',
      expDate: '',
      transactionSum: 0,
      terminalNumber: this.currentUser.terminalNumber,
      paymentPassword: this.currentUser.paymentPassword
    };
  }

  refreshData(): void {
    console.log('Attempting to refresh data...');
    
    // Clear the filtered data from the ReportService to ensure full data is loaded
    this.reportService.setFilteredData([]);  // Clear any filtered data
    
    // Reload the full data from the data service
    this.dataService.getAllData().subscribe(
      response => {
        console.log('Data fetched on refresh:', response);
        this.processData(response);  // This will update the table with unfiltered data
        this.loadFinancialSummary();  // Update financial summary
      },
      error => {
        console.error('Error fetching data on refresh:', error);
      }
    );
  }

  // Open the modal
  openAddDonationModal() {
    this.showModal = true;
    console.log('Modal Opened:', this.showModal);
  }

  // Close the modal
  closeAddDonationModal() {
    this.showModal = false;
    console.log('Modal Closed:', this.showModal);
  }

  // Handle the donation added event from the child component
  handleDonationAdded(success: boolean) {
    if (success) {
      alert('Donation added successfully!');
    } else {
      alert('Failed to add donation. Please try again.');
    }
    this.closeAddDonationModal();  // Close modal whether success or failure
  }
  
}
