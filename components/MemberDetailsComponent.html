import { ChangeDetectorRef, Component, OnInit } from '@angular/core';
import { MemberService } from '../member.service'; // Adjust the import path as necessary
import { UserService } from 'app/user.service';

@Component({
  selector: 'app-member-details',
  templateUrl: './member-details.component.html',
  styleUrls: ['./member-details.component.scss']
})
export class MemberDetailsComponent implements OnInit {
  members: any[] = [];
  currentUser: any;  // Renaming to differentiate from AppUser field


  constructor(private userService: UserService, private memberService: MemberService, private changeDetectorRef: ChangeDetectorRef) {}

  ngOnInit(): void {
    console.log('MemberDetailsComponent initialized');
    
    // Fetch the current user info
    this.userService.getCurrentUserInfo().subscribe(
      (userInfo) => {
        this.currentUser = userInfo;
        console.log('Current user info:', this.currentUser);
      },
      (error) => console.error('Error fetching current user info:', error)
    );

    // Fetch all members
    this.fetchAllMembers();
  }

  fetchAllMembers(): void {
    this.memberService.fetchMembersIfNeeded().subscribe(
      (members) => {
        this.members = members;
        this.changeDetectorRef.detectChanges(); // Ensure UI updates with new data
      },
      (error) => console.error('Error fetching members:', error)
    );
  }

  addMember(): void {
    console.log('Opening add member dialog...');
    // Assume modal is handled here
    this.changeDetectorRef.detectChanges(); // Ensure UI updates when modal content changes
  }

  handleMemberAdded(memberAdded: boolean): void {
    if (memberAdded) {
      this.fetchAllMembers(); // Refresh the list if new member is successfully added
      console.log('New member was added successfully.');
    } else {
      console.error('Failed to add new member.');
    }
  }
}
