import { HttpHeaders } from '@angular/common/http';
import { Component, OnInit } from '@angular/core';
import { AppUser } from 'app/models/app-user.model';
import { Role } from 'app/models/role.model';
import { UserService } from 'app/user.service';
import { catchError, Observable, tap } from 'rxjs';

@Component({
  selector: 'app-user-management',
  templateUrl: './user-management.component.html',
  styleUrls: ['./user-management.component.scss']
})
export class UserManagementComponent implements OnInit {
  users: AppUser[] = [];
  roles: Role[] = [];
  userSelectedRoles: { [userId: number]: Role[] } = {}; // Track roles per user
  selectedRoles: Role[] = []; // Array to store selected roles
  selectedExpendituresFile: File | null = null; // For expenditures Excel
  message: string = ''; // To hold success or error messages
  isSuccess: boolean = true; // To determine if the message is success or error
  selectedFile: File | null = null;

  newUser: AppUser = new AppUser(); // No need to manually define fields

  selectedUser: AppUser | null = null; // Selected user for role assignment

  constructor(private userService: UserService) {}

  ngOnInit() {
    this.loadRoles().subscribe(() => {
      this.loadUsers();
    });
    console.log('Roles available:', this.roles); // Make sure roles are loaded correctly
  }
  // isRoleSelected(user: AppUser, role: Role): boolean {
  //   const selectedRoles = this.userSelectedRoles[user.id] || [];
  //   const isSelected = selectedRoles.some(selectedRole => selectedRole.name === role.name);
    
  //   console.log(`Checking if role '${role.name}' is selected for user '${user.username}': ${isSelected}`);
  //   console.log('User selected roles:', selectedRoles);
    
  //   return isSelected;
  // }
  isRoleSelected(user: AppUser, role: Role): boolean {
    const userRoles = this.userSelectedRoles[user.id] || [];
  
    // Check if the role exists in the user's roles array (by Role.name)
    const isSelected = userRoles.some(userRole => userRole.name === role.name);    
    return isSelected;
  }
  
// Toggle role selection for a specific user
onRoleChange(event: any, user: AppUser, role: Role) {
  const isChecked = event.target.checked;

  // Initialize userSelectedRoles if not present
  if (!this.userSelectedRoles[user.id]) {
    this.userSelectedRoles[user.id] = [...user.roles]; // Initialize with user's existing roles
  }

  const userRoles = this.userSelectedRoles[user.id];
  const existingRoleIndex = userRoles.findIndex(selectedRole => selectedRole.name === role.name);

  if (isChecked) {
    // Add the role if it's checked
    if (existingRoleIndex === -1) {
      userRoles.push(role);
    }
  } else {
    // Remove the role if it's unchecked
    if (existingRoleIndex !== -1) {
      userRoles.splice(existingRoleIndex, 1);
    }
  }
  
  console.log(`Updated roles for user '${user.username}':`, userRoles);
}

// Assign roles to the selected user
updateUserRoles(user: AppUser) {
  const selectedRoles = this.userSelectedRoles[user.id] || [];

  console.log('Submitting roles for user:', user.username);
  console.log('Roles being submitted:', selectedRoles.map(r => r.name));

  const trimmedRoles = selectedRoles.map(r => r.name.trim()); // Trim the roles before submitting

  this.userService.assignRoles(user.id, trimmedRoles).subscribe({
    next: () => {
      this.loadUsers(); // Reload users to reflect changes
      this.message = 'Roles assigned successfully!';
      this.isSuccess = true;
      setTimeout(() => {
        this.message = '';
        // Reload the page to reflect the changes immediately
        window.location.reload();
      }, 3000); // Adjust the delay as needed
    },
    error: (err) => {
      this.message = err?.error?.message || err?.message || 'An error occurred while assigning roles.';
      this.isSuccess = false;
      setTimeout(() => {
        this.message = '';
      }, 3000);
    }
  });
}


// Load users and roles
loadUsers() {
  this.userService.getAllUsers().subscribe(users => {
    this.users = users;
    console.log('Loaded Users:', this.users); // Add this to check users

    // Ensure roles are loaded
    if (!this.roles || this.roles.length === 0) {
      console.error("Roles are not loaded properly. Unable to map roles to users.");
      return;
    }

    // Initialize userSelectedRoles map
    this.users.forEach(user => {
      // Check if user.roles contains strings or objects
      this.userSelectedRoles[user.id] = user.roles.map((userRole: string | Role) => {
        // Handle userRole if it's a string (role name)
        const roleName = typeof userRole === 'string' ? userRole : userRole.name;

        // Find the corresponding Role object by its name
        const matchingRole = this.roles.find(role => role.name.trim() === roleName.trim());

        // If the role is found, return it, otherwise return a default empty Role object
        return matchingRole || { id: 0, name: '', roleCode: 0 };
      });
    });

    // Log to ensure user roles are correctly initialized
    console.log('User roles mapped after load:', this.userSelectedRoles);
  });
}

loadRoles(): Observable<Role[]> {
  return this.userService.getRoles().pipe(
    tap(roles => {
      console.log('Roles fetched from server:', roles); // Log the raw roles from server

      if (roles && roles.length > 0) {
        // Log role names before trimming
        console.log('Raw role names before trimming:', roles.map(role => role.name));

        this.roles = roles.map(role => ({
          ...role,
          name: role.name ? role.name.trim() : ''  // Trimming the role names
        }));

        // Log role names after trimming
        console.log('Role names after trimming:', this.roles.map(role => role.name));
      } else {
        console.error('No roles available from the server.');
      }
    }),
    catchError(error => {
      // Log error if something goes wrong
      console.error('Error fetching roles:', error);
      return [];
    })
  );
}


 // Handle checkbox change for role selection
 onRoleCheckboxChange(event: any, role: Role) {
  const isChecked = event.target?.checked; // Check if the checkbox is checked

  if (isChecked) {
    // Add the role if it's checked and not already in selectedRoles
    if (!this.selectedRoles.some(r => r.id === role.id)) {
      this.selectedRoles.push(role);
      console.log('Role added:', role);
    }
  } else {
    // Remove the role if it's unchecked
    this.selectedRoles = this.selectedRoles.filter(r => r.id !== role.id);
    console.log('Role removed:', role);
  }

  // Log the updated state of selectedRoles for debugging
  console.log('Updated selected roles:', this.selectedRoles);
}


  // Toggle the selection of roles
  toggleRoleSelection(role: Role) {
    const index = this.selectedRoles.findIndex(r => r.id === role.id);
    if (index > -1) {
      this.selectedRoles.splice(index, 1); // Unselect if already selected
    } else {
      this.selectedRoles.push(role); // Select if not selected
    }
  }

  selectUserForRoleAssignment(user: AppUser) {
    this.selectedUser = user;
    this.selectedRoles = [...user.roles]; // Initialize selectedRoles with user's current roles
    console.log('Selected user for role assignment:', this.selectedUser);
    console.log('Pre-populated selectedRoles:', this.selectedRoles);
  }
 
  
  // Function to check if a user has a specific role
  hasRole(user: AppUser, roleName: string): boolean {
    return user.roles.some(role => role.name === roleName);
  }

  // Assign roles to the selected user
  assignRoles(user: AppUser) {
    console.log('Assigning roles for user:', user.username);
    console.log('Selected roles before assignment:', this.selectedRoles);
  
    const roleNames = this.selectedRoles.map(role => role.name); // Get role names
  
    // Send selected roles to backend
    this.userService.assignRoles(user.id, roleNames).subscribe({
      next: () => {
        this.loadUsers(); // Reload users to reflect changes
        this.message = 'Roles assigned successfully!';
        this.isSuccess = true;
        setTimeout(() => {
          this.message = '';
        }, 3000);
      },
      error: (err) => {
        this.message = err?.error?.message || err?.message || 'An error occurred while assigning roles.';
        this.isSuccess = false;
        setTimeout(() => {
          this.message = '';
        }, 3000);
      }
    });
  }
  

  // Add a new user with selected roles
  addUser() {
    this.newUser.roles = this.selectedRoles;
    this.userService.registerUser(this.newUser).subscribe({
      next: () => {
        this.loadUsers(); // Reload users after adding a new one
        this.selectedRoles = []; // Reset the selected roles
        this.message = 'User added successfully!';
        this.isSuccess = true;
        this.resetForm();
        setTimeout(() => {
          this.message = ''; // Clear message after a few seconds
        }, 3000);
      },
      error: (err) => {
        this.message = err?.error?.message || err?.message || 'An error occurred during user creation.';
        this.isSuccess = false;
        setTimeout(() => {
          this.message = '';
        }, 3000);
      }
    });
  }

  // Reset the form after adding the user
  resetForm() {
    this.newUser = {
      id: 0,
      username: '',
      email: '',
      password: '',
      roles: [],
      clientId: null,
      apiKey: null,
      apiSecret: null
    };
  }

  // Get role names as a comma-separated string
  getRoleNames(user: AppUser): string {
    return user.roles.map(role => role.name).join(', ');
  }

  // File selection event handler
  onFileSelected(event: any) {
    const file: File = event.target.files[0];
    if (file) {
      this.selectedFile = file;
    }
  }

  // Upload the Excel file
  uploadExcel() {
    if (this.selectedFile) {
      this.userService.uploadExcelFile(this.selectedFile).subscribe({
        next: (response) => {
          this.message = 'File uploaded successfully!';
          this.isSuccess = true;
        },
        error: (error) => {
          this.message = 'Failed to upload file.';
          this.isSuccess = false;
        }
      });
    } else {
      this.message = 'No file selected.';
      this.isSuccess = false;
    }
  }

  // Upload expenditures Excel file
  uploadExpendituresExcel() {
    if (this.selectedExpendituresFile) {
      this.userService.uploadExpendituresExcelFile(this.selectedExpendituresFile).subscribe({
        next: () => {
          this.message = 'Expenditures Excel file uploaded successfully';
          this.isSuccess = true;
        },
        error: (err) => {
          this.message = err?.error?.message || err?.message || 'An error occurred during expenditures file upload.';
          this.isSuccess = false;
        }
      });
    }
  }

  onExpendituresFileSelected(event: any) {
    this.selectedExpendituresFile = event.target.files[0];
  }
}
